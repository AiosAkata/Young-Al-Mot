#테이블 생성 전에 yam데이터베이스 생성
create database yam;

#user 테이블
create table user(
   `user_no` int NOT NULL auto_increment,
   `user_id` varchar(20) NOT NULL,
   `password` varchar(20) NOT NULL,
   `user_name` varchar(20) NOT NULL,
   `email` varchar(40) NOT NULL,
   primary key(`user_no`)
);
